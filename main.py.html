import time
from tkinter import *

def timer():

    instance = int(input.get())
    for i in range(instance,0,-1):
        print(i)
        countdown["text"] = i
        window.update()
        time.sleep(1)
    print("TIMES UP!")
    countdown["text"] = ("Times Up!")
    window.update()


window = Tk()
window.title("Timer")
window.config(padx=20, pady=20)

time_label = Label(text ="Please enter time in seconds ", font=("Arial",12) )
time_label.pack()

input = Entry(width=10)
input.pack()

start_button = Button(text="Start", command=timer)
start_button.pack()

countdown = Label(text=0, font=("Arial",12))
countdown.pack()
